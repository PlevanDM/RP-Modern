# Быстрый деплой RepairHub Pro на сервер
# Скопируйте и выполните эти команды на сервере

cd /root

# Обновление и установка зависимостей
apt update -y
apt install -y curl git build-essential

# Установка Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | bash -
apt install -y nodejs

# Клонирование репозитория
if [ -d "repairhub" ]; then
    cd repairhub && git pull origin 24.10v3
else
    git clone -b 24.10v3 https://github.com/PlevanDM/RP-Modern.git repairhub
    cd repairhub
fi

# Установка зависимостей и сборка
npm install
npm run build

# Установка PM2
npm install -g pm2

# Запуск приложения
pm2 delete repairhub 2>/dev/null || true
pm2 start npm --name "repairhub" -- start -- --host 0.0.0.0 --port 3000
pm2 save
pm2 startup systemd -u root --hp /root || true

echo "✅ Деплой завершен! Приложение доступно на http://64.176.72.139:3000"
pm2 status
